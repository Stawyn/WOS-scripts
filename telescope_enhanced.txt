-- Enhanced Telescope Interface with Manual Coordinate Selection
-- Allows users to input specific coordinates, use presets, scan sequentially, and view history
-- 
-- Requirements:
-- • Screen connected to Port 1
-- • Keyboard connected to Port 1  
-- • At least one Telescope connected
-- • Optional: Modem for external logging

local Telescopes = GetParts("Telescope")
local Screen = GetPartFromPort(1, "Screen")
local Keyboard = GetPartFromPort(1, "Keyboard")
local Modem = GetPart("Modem")

-- Verificação de componentes essenciais
if not Screen then
    error("❌ Screen não encontrado na porta 1! Conecte um Screen na porta 1.")
end

if not Keyboard then
    print("⚠️ Keyboard não encontrado na porta 1. Entrada manual será limitada.")
end

if not Telescopes or #Telescopes == 0 then
    error("❌ Nenhum telescópio encontrado! Conecte pelo menos um telescópio.")
end

print("✅ Componentes verificados:")

local MIN_COORD = -100
local MAX_COORD = 100

-- Estado da aplicação

local coordinateHistory = {}
local isSequentialMode = false
local sequentialPosition = 0

    {name = "Sector A", x1 = 25, x2 = 25, y1 = 25, y2 = 25},
    {name = "Sector B", x1 = -25, x2 = -25, y1 = 25, y2 = 25},

}

-- Coordenadas favoritas (podem ser adicionadas pelo usuário)
local favoriteCoordinates = {}

-- Referências dos elementos UI
local UI = {}
local activeInputField = nil

-- Funções utilitárias
local function ValidateCoordinate(value)
    local num = tonumber(value)
    if not num then return false end
    return num >= MIN_COORD and num <= MAX_COORD
end

local function SaveToHistory(coords)
    local entry = {
        x1 = coords.x1, x2 = coords.x2, y1 = coords.y1, y2 = coords.y2,
        timestamp = os.time(),
        name = string.format("(%d,%d,%d,%d)", coords.x1, coords.x2, coords.y1, coords.y2)
    }
    
    -- Evita duplicatas recentes
    for i, item in ipairs(coordinateHistory) do
        if item.x1 == coords.x1 and item.x2 == coords.x2 and 
           item.y1 == coords.y1 and item.y2 == coords.y2 then
            table.remove(coordinateHistory, i)
            break
        end
    end
    
    table.insert(coordinateHistory, 1, entry)
    
    -- Mantém apenas os últimos 10 registros
    if #coordinateHistory > 10 then
        coordinateHistory[11] = nil
    end
    
    UpdateHistoryDisplay()
end

local function ConfigureTelescope(coords)
    if not Telescopes or #Telescopes == 0 then
        UI.statusLabel.Text = "❌ Nenhum telescópio encontrado!"
        return false
    end
    
    local success, result = pcall(function()
        Telescopes[1]:Configure({ 
            ViewCoordinates = string.format("%d,%d,%d,%d,false", 
                coords.x1, coords.x2, coords.y1, coords.y2) 
        })
    end)
    
    if success then
        currentCoordinates = coords
        UI.statusLabel.Text = string.format("✅ Visualizando: (%d,%d,%d,%d)", 
            coords.x1, coords.x2, coords.y1, coords.y2)
        SaveToHistory(coords)
        UpdateCurrentDisplay()
        
        -- Tenta obter informações do planeta
        task.wait(0.5)
        GetPlanetInfo()
        return true
    else
        UI.statusLabel.Text = "❌ Erro ao configurar telescópio"
        return false
    end
end

local function GetPlanetInfo()
    if not Telescopes or #Telescopes == 0 then return end
    
    local success, planetData = pcall(function()
        return Telescopes[1]:GetCoordinate()
    end)
    
    if success and planetData then
        local planetName = planetData.Name or planetData.name or planetData.PlanetName or "Desconhecido"
        if planetName and planetName ~= "" and planetName ~= "SECTOR () ()" then
            UI.planetInfoLabel.Text = string.format("🌎 Planeta: %s", planetName)
            if planetData.Type then
                UI.planetInfoLabel.Text = UI.planetInfoLabel.Text .. string.format("\n📍 Tipo: %s", planetData.Type)
            end
        else
            UI.planetInfoLabel.Text = "🌌 Espaço vazio"
        end
    else
        UI.planetInfoLabel.Text = "❓ Verificando..."
    end
end

-- Função sequencial (adaptada de s3.txt)
local function PositionToCoordinates(position)
    local COORD_RANGE = MAX_COORD - MIN_COORD + 1
    local TOTAL_COMBINATIONS = COORD_RANGE ^ 4
    
    if position >= TOTAL_COMBINATIONS then
        return nil
    end
    
    local temp = position
    local x1 = MIN_COORD + (temp % COORD_RANGE)
    temp = math.floor(temp / COORD_RANGE)
    
    local x2 = MIN_COORD + (temp % COORD_RANGE)
    temp = math.floor(temp / COORD_RANGE)
    
    local y1 = MIN_COORD + (temp % COORD_RANGE)
    temp = math.floor(temp / COORD_RANGE)
    
    local y2 = MIN_COORD + (temp % COORD_RANGE)
    
    return {x1 = x1, x2 = x2, y1 = y1, y2 = y2, position = position}
end

local function NextSequentialCoordinates()
    local coords = PositionToCoordinates(sequentialPosition)
    if coords then
        sequentialPosition = sequentialPosition + 1
        return coords
    end
    return nil
end

        BackgroundColor3 = Color3.fromRGB(0, 150, 0),

function UpdateCurrentDisplay()
    UI.currentCoordsLabel.Text = string.format("📍 Coordenadas: (%d,%d,%d,%d)", 
        currentCoordinates.x1, currentCoordinates.x2, currentCoordinates.y1, currentCoordinates.y2)
    
    -- Atualiza campos de entrada
    for key, field in pairs(UI.inputFields) do
        field.Text = tostring(currentCoordinates[key])
    end
end

function UpdateHistoryDisplay()
    -- Limpa histórico existente
    for _, child in pairs(UI.historyScroll:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end
    
    -- Adiciona entradas do histórico
    for i, entry in ipairs(coordinateHistory) do
        local historyButton = Screen:CreateElement("TextButton", {
            Size = UDim2.new(1, -10, 0, 20),
            Position = UDim2.new(0, 5, 0, (i - 1) * 25),
            Text = entry.name,
            TextColor3 = Color3.fromRGB(255, 255, 255),
            TextScaled = true,
            BackgroundColor3 = Color3.fromRGB(60, 60, 60),
            BorderSizePixel = 1,
            Parent = UI.historyScroll
        })
        
        historyButton.MouseButton1Click:Connect(function()
            ConfigureTelescope(entry)
        end)
    end
    
    UI.historyScroll.CanvasSize = UDim2.new(0, 0, 0, #coordinateHistory * 25)
end

-- Conecta entrada de teclado
if Keyboard then
    Keyboard.TextInputted:Connect(function(text, player)
        if activeInputField then
            local input = text:sub(1, -2) -- Remove quebra de linha
            if ValidateCoordinate(input) then
                activeInputField.field.Text = input
                activeInputField.field.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
                activeInputField = nil
                UI.statusLabel.Text = "✅ Coordenada atualizada"
            else
                UI.statusLabel.Text = string.format("❌ Valor inválido! Use números de %d a %d", 
                    MIN_COORD, MAX_COORD)
            end
        end
    end)
else
    print("⚠️ Entrada de teclado não disponível. Use apenas os botões predefinidos.")
end

-- Scanner sequencial automático
task.spawn(function()
    while true do
        if isSequentialMode then
            task.wait(3) -- Aguarda 3 segundos entre coordenadas
            local coords = NextSequentialCoordinates()
            if coords then
                ConfigureTelescope(coords)
            else
                isSequentialMode = false
                UI.sequentialButton.Text = "▶️ INICIAR SEQUENCIAL"
                UI.sequentialButton.BackgroundColor3 = Color3.fromRGB(150, 0, 150)
                UI.statusLabel.Text = "⚠️ Scanner sequencial completado"
            end
        else
            task.wait(1)
        end
    end
end)

-- Inicialização
print("🔭 Carregando Telescópio Aprimorado...")
CreateUI()
UpdateCurrentDisplay()
UpdateHistoryDisplay()

print("   🔄 Scanner sequencial automático")
